            {/* Event Details Tab */}
            {activeTab === 'event' && (
              <div className="space-y-6 py-6">
                {/* Venue and Space Selection - TOP (Already selected from calendar) */}

                  <div className="flex items-center gap-2 mb-3">
                    <Building2 className="size-5 text-blue-600" />

                    {/* Venue Selection - LOCKED/READ-ONLY when coming from calendar */}

                        Venue

                        <SelectContent>
                          {selectedPrimeSpace?.subSpaces.map((subSpace) => (
